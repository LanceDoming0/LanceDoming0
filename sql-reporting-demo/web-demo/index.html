EXAMPLE QUERY

Relational project report

SELECT c.client_name, p.project_name,
       p.status, p.progress_percent,
       p.budget, p.due_date
FROM projects AS p
JOIN clients AS c
  ON c.client_id = p.client_id
ORDER BY p.due_date;
