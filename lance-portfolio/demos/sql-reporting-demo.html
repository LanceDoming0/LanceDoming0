Sample SQL reporting query

SELECT c.client_name, c.service_type, c.status,
       p.project_name, p.progress_percent, p.budget
FROM clients c
LEFT JOIN projects p ON p.client_id = c.client_id
ORDER BY p.progress_percent DESC;
